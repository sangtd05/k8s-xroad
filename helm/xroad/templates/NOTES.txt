3. Access the services:

{{- if .Values.centralServer.enabled }}
   Central Server Admin Interface:
   {{- if contains "NodePort" .Values.centralServer.service.type }}
   export NODE_IP=$(kubectl get nodes -o jsonpath="{.items[0].status.addresses[0].address}")
   echo "https://$NODE_IP:{{ .Values.centralServer.service.adminNodePort }}"
   {{- else if contains "LoadBalancer" .Values.centralServer.service.type }}
   export SERVICE_IP=$(kubectl get svc {{ include "xroad.centralServer.fullname" . }} -o jsonpath="{.status.loadBalancer.ingress[0].ip}")
   echo "https://$SERVICE_IP:{{ .Values.centralServer.ports.admin }}"
   {{- else }}
   kubectl port-forward svc/{{ include "xroad.centralServer.fullname" . }} {{ .Values.centralServer.ports.admin }}:{{ .Values.centralServer.ports.admin }}
   echo "https://localhost:{{ .Values.centralServer.ports.admin }}"
   {{- end }}
{{- end }}

{{- if .Values.securityServer.enabled }}
   Security Server Admin Interface:
   {{- if contains "NodePort" .Values.securityServer.service.type }}
   export NODE_IP=$(kubectl get nodes -o jsonpath="{.items[0].status.addresses[0].address}")
   echo "https://$NODE_IP:{{ .Values.securityServer.service.adminNodePort }}"
   {{- else if contains "LoadBalancer" .Values.securityServer.service.type }}
   export SERVICE_IP=$(kubectl get svc {{ include "xroad.securityServer.fullname" . }} -o jsonpath="{.status.loadBalancer.ingress[0].ip}")
   echo "https://$SERVICE_IP:4000"
   {{- else }}
   kubectl port-forward svc/{{ include "xroad.securityServer.fullname" . }} 4000:4000
   echo "https://localhost:4000"
   {{- end }}
{{- end }}
